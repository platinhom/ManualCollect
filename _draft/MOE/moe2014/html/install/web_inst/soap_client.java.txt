// ============================================================================
// Sample program to call a MOE/Web SOAP service. It requires the wsimport
// program to pre generate a xxx.moeweb.moewebserver package. For example from
// Linux command line, do the following steps:
// 1) mkdir generated
// 2) wsimport -d generated -keep http://{host}:{port}/MOEWebServer.wsdl
// 3) javac -cp generated soap_client.java
// 4) java -cp ./:generated soap_client
// Tested with Oracle Java 1.7.0_40 on Linux.
// This program is distributed in the hope that it will be useful, but
// WITHOUT ANY WARRANTY.
//=============================================================================

import xxx.moeweb.moewebserver.*; // replace it with the wsimport generated name
import java.io.*;
import java.util.*;

public class soap_client {

public static void main(String[] args)    {

    // Set up server endpoint to which the client is going to connect.

    MOEWebServerPortType soapclient = new MOEWebService().getMOEWebServerPort();

    try {

	// Set item values in the SOAP request.

	SMILESToMolInputType input = new SMILESToMolInputType();
	input.setSmiles("c(c)c");
	input.setV3000(false);

	// Send SOAP request and wait for result.
	// Returned XML is parsed into an Java object for
	// easy access to the values contained in SOAP Response.

	SMILESToMolOutputType output = soapclient.smilesToMol(input);

	// Print out returned molecule data.

	System.out.println("Result:\n" + output.getMolecule());

    } catch(Exception ex) {

	// Report error information.

	ex.printStackTrace();
    }
 

}

}
